#include <stdio.h>
#include <malloc.h>
typedef struct node
{
int data;
struct node* next;
}snode, * ptr;
ptr creat();
void output(ptr p);
ptr insert(ptr head, int x);
void destory(ptr q);
int main()
{
ptr head;
ptr headn;
head = creat();
headn = creat();
int x;
while (scanf(“%d”, &x) == 1)
{
if (x > 0)
head = insert(head, x);
else
headn = insert(headn, x);
}
output(head);
printf(“\n”);
output(headn);
destory(head);
destory(headn);
return 0;
}
ptr creat()
{
ptr head = malloc(sizeof(snode));
head->next = NULL;
return head;
}
ptr insert(ptr head, int x)
{
if (head->next == NULL)
{
ptr t = malloc(sizeof(snode));
t->data = x;
t->next = NULL;
head->next = t;
}
ptr rear = head;
ptr p = head->next;
while (p != NULL)
{
if (x < p->data)
{
ptr t = malloc(sizeof(snode));
t->data = x;
t->next = NULL;
t->next = rear->next;
rear->next = t;
return head;
}
p = p->next;
rear = rear->next;
}
if (x > rear->data)
{
ptr t = malloc(sizeof(snode));
t->data = x;
t->next = NULL;
rear->next = t;
}
return head;
}
void output(ptr p)
{
ptr cur = p->next;
while (cur->next != NULL)
{
printf(“%d->”, cur->data);
cur = cur->next;
}
printf(“%d”, cur->data);
}
void destory(ptr q)
{
ptr L = q;
ptr h = q->next;
while (h->next)
{
L->next = h->next;
free(h);
h = L->next;
}
free(L);
}

#include <stdio.h>
#define INF 0x3f3f3f3f
int dis[2505], vis[2505], head[2505], cnt;
int n, m, s, t;
void add(int a, int b, int c);
void swap(int x, int y);
void push(int w, int id);
void pop();
void dijstra(int s);
typedef struct {
int v, w, nxt;
} Edge;
Edge edge[6205];
typedef struct {
int val, id;
} HNode;
typedef struct {
HNode h[6205];
int size;
} Heap;
Heap h;
int main()
{
int i, j;
scanf(“%d%d%d%d”, &n, &m, &s, &t);
for(i=1; i<=m; ++i)
{
int a, b, c;
scanf(“%d%d%d”, &a, &b, &c);
add(a, b, c);
add(b, a, c);
}
dijstra(s);
printf(“%d\n”, dis[t]);
return 0;
}
void add(int a, int b, int c)
{
edge[++cnt].v = b;
edge[cnt].w = c;
edge[cnt].nxt = head[a];
head[a] = cnt;
return;
}
void swap(int x, int y)
{
HNode t = h.h[x];
h.h[x] = h.h[y];
h.h[y] = t;
return;
}
void push(int w, int id)
{
++h.size;
h.h[h.size].val = w;
h.h[h.size].id = id;
int i;
for(i=h.size; (i>>1)&&(h.h[i>>1].val > h.h[i].val); i>>=1)
swap(i>>1, i);
return;
}
void pop()
{
h.h[1] = h.h[h.size];
–h.size;
int i, mn, flag;
for(i=1; (i<<1)<=h.size; )
{
flag = 1;
mn = h.h[i].val < h.h[i<<1].val ? i : i<<1;
if((i<<1|1)<=h.size && h.h[i<<1|1].val < h.h[mn].val)
mn = i<<1|1;
if(mn != i) swap(mn, i);
else break;
i = mn;
}
return;
}
void dijstra(int s)
{
int i, j;
for(i=1; i<=n; ++i)
dis[i] = INF;
dis[s] = 0;
push(0, s);
while(h.size)
{
int u = h.h[1].id;
if(u == t) return;
if(vis[u])
{
pop();
continue;
}
vis[u] = 1;
pop();
for(i=head[u]; i; i=edge[i].nxt)
{
int v = edge[i].v, w = edge[i].w;
if(dis[u]+w < dis[v])
{
dis[v] = dis[u]+w;
push(dis[v], v);
}
}
}
return;
}
